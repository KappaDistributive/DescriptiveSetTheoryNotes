\begin{lemma}
  Suppose $A \subseteq ^{\omega}\omega$ has a $\kappa$-semiscale. Then
  $A$ is $\kappa$-Souslin.
\end{lemma}

\begin{proof}
  Let $(\phi_n \mid n < \omega)$ be a $\kappa$-semiscale on $A$. Let
  \[
    T = \{ (x \restriction n, (\phi_0(x), \phi_1(x), \ldots,
    \phi_{n-1}(x)) \mid x \in A \}.
  \]
  Clearly $T$ is a tree and $A \subseteq p[T]$. On the other hand, if
  $x \in p[T]$, fix $f \colon \omega \to \kappa$ such that
  $(x,f) \in T$. For each $n < \omega$ there is now some $a_n \in A$
  of length $A$ such that
  $(x\restriction n, f \restriction n) = (a_n \restriction n,
  (\phi_0(a_n), \ldots, \phi_{n-1}(a_{n}))$. We have
  $\lim_{n < \omega}a_n = x$. Furthermore, for all $m < n < l < \omega$,
  \[
   \phi_m(a_n) = \phi_m(a_l),
 \]
 i.e. $(\phi_m(a_n) \mid n < \omega)$ is eventually constant and thus
 $x \in A$.
\end{proof}
